# Push-Wake Implementation Plan

> ŌÜĀ’ĖÅ **SUPERSEDED 2026-04-17** ŌĆö do NOT execute. The spec this plan derives from
> (`docs/superpowers/specs/2026-04-16-push-wake-design.md`) was rejected after parallel-cli research revealed (1) `dispatchMessage` pulls Safari to foreground on every call, and (2) Apple's documented workaround is to switch to event-page background (`scripts` + `persistent:false`). The new direction is captured in `CHECKPOINT.md` and the forward-carry prompt. This plan remains as historical record only.

---

> **For agentic workers:** REQUIRED SUB-SKILL: Use the `upp:executing-plans` skill to implement this plan task-by-task. Supports two modes: subagent-driven (recommended, fresh subagent per task with three-stage review) or inline execution. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Wake dormant Safari MV3 service workers via `SFSafariApplication.dispatchMessage` from a LaunchAgent-managed containing app, with keepalive-alarm safety net, so `extension_execute` roundtrips succeed cold (no user `.app` reopen required).

**Architecture:** Daemon's `AppRelayServer` (TCP:19475) broadcasts a `"wake\n"` byte to the containing app whenever a command is queued. `AppDelegate` in the containing app (running as LaunchAgent, `LSUIElement: YES`) receives the wake and calls `SFSafariApplication.dispatchMessage(name: "commandReady")`. Extension's `background.js` registers a `runtime.onMessage` branch that triggers `pollLoop` on wake. Task #3 daemon long-polling's fast-path delivers the queued command on the first subsequent poll. The 1-minute keepalive alarm stays as a defense-in-depth fallback.

**Tech Stack:** Swift 6 (daemon + appex + containing app), Network.framework (NWConnection, NWListener), Safari Services (`SFSafariApplication.dispatchMessage`), JavaScript (Safari WebExtension background.js), launchd (`launchctl bootstrap`), vitest + custom Swift test harness.

**Spec:** `docs/superpowers/specs/2026-04-16-push-wake-design.md`

```swift
import Foundation
import Network

/// Listens on TCP port 19475 for the containing app's relay connection.
/// When signalWake() is called (command queued), sends a wake byte to all
/// connected relay clients. The app receives this and calls
/// SFSafariApplication.dispatchMessage() to wake the extension.
public final class AppRelayServer: @unchecked Sendable {

    private let listener: NWListener
    private let queue = DispatchQueue(label: "com.safari-pilot.app-relay", qos: .userInitiated)
    private var clients: [NWConnection] = []
    private var actualPort: UInt16 = 0

    public init(port: UInt16 = 19475) {
        let params = NWParameters.tcp
        params.allowLocalEndpointReuse = true
        do {
            self.listener = try NWListener(using: params, on: NWEndpoint.Port(rawValue: port)!)
        } catch {
            Logger.error("AppRelayServer: failed to create listener on port \(port): \(error)")
            self.listener = try! NWListener(using: .tcp)
        }
    }

    @discardableResult
    public func start() -> UInt16? {
        let semaphore = DispatchSemaphore(value: 0)
        var startedPort: UInt16?

        listener.stateUpdateHandler = { [weak self] state in
            switch state {
            case .ready:
                if let port = self?.listener.port?.rawValue {
                    self?.actualPort = port
                    startedPort = port
                    Logger.info("AppRelayServer listening on localhost:\(port)")
                }
                semaphore.signal()
            case .failed(let error):
                Logger.error("AppRelayServer failed to start: \(error)")
                semaphore.signal()
            default:
                break
            }
        }

        listener.newConnectionHandler = { [weak self] connection in
            self?.acceptClient(connection)
        }

        listener.start(queue: queue)
        _ = semaphore.wait(timeout: .now() + 5)
        return startedPort
    }

    public func stop() {
        queue.sync {
            for client in clients { client.cancel() }
            clients.removeAll()
        }
        listener.cancel()
        Logger.info("AppRelayServer stopped")
    }

    public func signalWake() {
        // Implemented in Task 2
        fatalError("signalWake not yet implemented")
    }

    public var clientCount: Int {
        queue.sync { clients.count }
    }

    private func acceptClient(_ connection: NWConnection) {
        connection.stateUpdateHandler = { [weak self] state in
            switch state {
            case .ready:
                Logger.info("AppRelayServer: app relay client connected")
            case .cancelled, .failed:
                self?.queue.async {
                    self?.clients.removeAll { $0 === connection }
                }
            default:
                break
            }
        }
        connection.start(queue: queue)
        queue.async {
            self.clients.append(connection)
        }
    }
}
```

```swift
import Foundation
import SafariServices

/// Thin indirection over SFSafariApplication.dispatchMessage so AppDelegate
/// can be tested without invoking real Safari infrastructure.
public protocol DispatchMessageService {
    func dispatch(name: String, toExtensionWithIdentifier extensionId: String, userInfo: [String: Any]?)
}

/// Production implementation ŌĆö delegates to SFSafariApplication.
public final class ProductionDispatcher: DispatchMessageService {
    public init() {}

    public func dispatch(name: String, toExtensionWithIdentifier extensionId: String, userInfo: [String: Any]?) {
        SFSafariApplication.dispatchMessage(
            withName: name,
            toExtensionWithIdentifier: extensionId,
            userInfo: userInfo
        ) { error in
            if let error = error {
                NSLog("Safari Pilot App: dispatchMessage error: %@", String(describing: error))
            }
        }
    }
}
```

```swift
import Cocoa
import SafariServices
import Network

let extensionBundleIdentifier = "com.safari-pilot.app.Extension"

@main
class AppDelegate: NSObject, NSApplicationDelegate {

    private var connection: NWConnection?
    private let queue = DispatchQueue(label: "com.safari-pilot.app-relay-client")
    private let dispatcher: DispatchMessageService

    // Production initializer ŌĆö uses real SFSafariApplication via ProductionDispatcher.
    override init() {
        self.dispatcher = ProductionDispatcher()
        super.init()
    }

    // Test initializer ŌĆö injected dispatcher.
    init(dispatcher: DispatchMessageService) {
        self.dispatcher = dispatcher
        super.init()
    }

    func applicationDidFinishLaunching(_ notification: Notification) {
        NSLog("Safari Pilot App: launched, LSUIElement=YES, connecting to daemon relay")
        connectToDaemon()
    }

    func applicationShouldTerminateAfterLastWindowClosed(_ sender: NSApplication) -> Bool {
        return false // stay alive even if the (invisible) window closes
    }

    private func connectToDaemon() {
        let host = NWEndpoint.Host("127.0.0.1")
        let port = NWEndpoint.Port(rawValue: 19475)!
        let conn = NWConnection(host: host, port: port, using: .tcp)
        connection = conn

        conn.stateUpdateHandler = { [weak self] state in
            switch state {
            case .ready:
                NSLog("Safari Pilot App: connected to daemon relay port 19475")
                self?.listenForWakeSignals()
            case .failed, .cancelled:
                NSLog("Safari Pilot App: relay connection dropped, retrying in 5s")
                self?.connection = nil
                DispatchQueue.main.asyncAfter(deadline: .now() + 5) {
                    self?.connectToDaemon()
                }
            default:
                break
            }
        }
        conn.start(queue: queue)
    }

    private func listenForWakeSignals() {
        connection?.receive(minimumIncompleteLength: 1, maximumLength: 65536) { [weak self] data, _, _, error in
            guard let self = self else { return }
            if let data = data, !data.isEmpty {
                NSLog("Safari Pilot App: wake signal received (\(data.count) bytes) ŌĆö dispatching to extension")
                self.dispatcher.dispatch(
                    name: "commandReady",
                    toExtensionWithIdentifier: extensionBundleIdentifier,
                    userInfo: nil
                )
                self.listenForWakeSignals() // continue listening
            } else if let error = error {
                NSLog("Safari Pilot App: receive error: \(error); reconnecting")
                self.connection?.cancel()
            }
        }
    }
}
```
